<script>

let records = [];

window.onload = function(){

    const saved = localStorage.getItem("records");

    if(saved){
        records = JSON.parse(saved);
        displayRecords();
    }

}

function insertRecord(){

    const first = document.getElementById("firstName").value;
    const middle = document.getElementById("middleName").value;
    const last = document.getElementById("lastName").value;
    const age = document.getElementById("age").value;

    if(first === "" || last === "" || age === ""){
        alert("Please fill required fields");
        return;
    }

    const record = {
        firstName:first,
        middleName:middle,
        lastName:last,
        age:age
    };

    records.push(record);

    displayRecords();
}

function displayRecords(){

    const table = document.getElementById("recordsTable");
    table.innerHTML = "";

    records.forEach((rec,index)=>{

        table.innerHTML += `
        <tr>
        <td>${rec.firstName}</td>
        <td>${rec.middleName}</td>
        <td>${rec.lastName}</td>
        <td>${rec.age}</td>
        <td>
        <button onclick="editRecord(${index})">Edit</button>
        <button onclick="deleteRecord(${index})">Delete</button>
        </td>
        </tr>
        `;

    });

}

function deleteRecord(index){

    records.splice(index,1);

    displayRecords();
}

function editRecord(index){

    const rec = records[index];

    document.getElementById("firstName").value = rec.firstName;
    document.getElementById("middleName").value = rec.middleName;
    document.getElementById("lastName").value = rec.lastName;
    document.getElementById("age").value = rec.age;

    // remove the original entry so it can be replaced when inserted again
    records.splice(index,1);

    displayRecords();

    // notify the user that editing is in progress
    alert("Record loaded for editing - make changes then click Insert");
}

function clearFields(){

    document.getElementById("firstName").value="";
    document.getElementById("middleName").value="";
    document.getElementById("lastName").value="";
    document.getElementById("age").value="";
}

function clearRecords(){

    records = [];
    displayRecords();
}

function sortRecords(){

    const field = document.getElementById("sortField").value;
    const order = document.getElementById("sortOrder").value;

    records.sort(function(a,b){
        let valA = a[field];
        let valB = b[field];

        if(field === "age"){
            // numeric comparison for age
            valA = Number(valA);
            valB = Number(valB);
        }else{
            // string comparison should ignore case
            valA = valA.toString().toLowerCase();
            valB = valB.toString().toLowerCase();
        }

        if(order === "az"){
            if(valA < valB) return -1;
            if(valA > valB) return 1;
            return 0;
        }else{ // za
            if(valA < valB) return 1;
            if(valA > valB) return -1;
            return 0;
        }
    });

    displayRecords();
}

function saveToLocalStorage(){

    localStorage.setItem("records", JSON.stringify(records));

    alert("Records saved to Local Storage");

}
